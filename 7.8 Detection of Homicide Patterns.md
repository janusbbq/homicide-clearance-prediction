(2) Clean Data (fast version)

# Remove invalid values

df = df[df["VICAGE"] != 999]   # unknown age
df = df[df["OFFAGE"] != 999]

# Keep only needed columns

cols = ["SOLVED", "VICAGE", "VICSEX", "VICRACE", "WEAPON", "YEAR"]
df = df[cols]

# Drop missing

df = df.dropna()

------------------------------------------------------------

(3) Encode Categorical Variables
df = pd.get_dummies(df, columns=["VICSEX", "VICRACE", "WEAPON"], drop_first=True)

------------------------------------------------------------

(4) Define X and y
X = df.drop("SOLVED", axis=1)
y = df["SOLVED"]

------------------------------------------------------------

(5) Train Model (fast + safe)
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

model = LogisticRegression(max_iter=1000)
model.fit(X_train, y_train)

------------------------------------------------------------

(6) Evaluate
from sklearn.metrics import accuracy_score

pred = model.predict(X_test)
print("Accuracy:", accuracy_score(y_test, pred))
